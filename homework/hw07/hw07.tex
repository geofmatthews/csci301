\noindent Find $LL(1)$ parse tables for each of the grammars 
below.
\begin{enumerate}
  
\item \label{llfirst}
  \begin{eqnarray*}
    S &\rightarrow& a \mid ba \mid c
  \end{eqnarray*}
\item
  \begin{eqnarray*}
    S &\rightarrow& A \mid B \mid C\\
    A &\rightarrow& aa\\
    B &\rightarrow& ba\\
    C &\rightarrow& ca\\
  \end{eqnarray*}
  
\item
  \begin{eqnarray*}
    S &\rightarrow& aa \mid bA\\
    A &\rightarrow& a \mid b
    \end{eqnarray*}

\item \label{lllast}
  \begin{eqnarray*}
    S &\rightarrow& aaSb \mid c
    \end{eqnarray*}


\end{enumerate}
